This is my didIWin function //
It shows if Player 1 or Player 2 has sunk all of the opposing ships //
Whenever a a place on the board is guessed, the space is replaced with the word "Sunk." The code then counts to see if the string "Sunk" appears 3 times //
BOARD_PLAYER_1 = [5][5]
BOARD_PLAYER_2 = [5][5]
DEF DIDIWIN():
    WIN_FACTOR = False
    loop ROW in BOARD_PLAYER_1[5][5]
        if ROW.COUNT("SUNK") == 3 then
            WIN_FACTOR = TRUE
      end if
end loop
    loop ROW in BOARD_PLAYER_2[5][5]
        if ROW.COUNT("SUNK") == 3 then
            WIN_FACTOR = TRUE
      end if
end loop
